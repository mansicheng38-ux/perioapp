// --- 核心：語音對照表，處理廣東話聽錯做英文嘅情況 --- function fixCantonesePhonetics(text) { let fixed = text.toLowerCase(); // 解決常見誤判：將聽錯嘅英文轉返做廣東話數字 const phoneticMap = { 'yellow': '16', // 「一六」最常被聽錯做 yellow 'hello': '16', // 另一種常見誤判 'sixteen': '16', 'low': '6', 'chat': '7', // 「七」常被聽成 chat 'easy': '2', // 「二」常被聽成 easy 'say': '4' // 「四」常被聽成 say }; for (let key in phoneticMap) { fixed = fixed.replace(new RegExp(key, 'g'), phoneticMap[key]); } // 處理中文數字 const cnMap = {'一':1,'二':2,'三':3,'四':4,'五':5,'六':6,'七':7,'八':8,'九':9,'零':0}; for (let key in cnMap) { fixed = fixed.replace(new RegExp(key, 'g'), cnMap[key]); } // 修正廣東話特有聽錯字 fixed = fixed.replace(/路|落|錄/g, '6').replace(/衣|伊|醫/g, '1'); return fixed; } // 修改後的 processData function function processData(text) { let processed = fixCantonesePhonetics(text); console.log("修正後文字:", processed); // 抓取所有數字 (例如 16, 17) const nums = processed.match(/\d{2}/g); // 廣東話缺失關鍵字 const isMissing = /缺失|冇|唔見|甩|蚊|missing|none/.test(processed); if (isMissing && nums) { nums.forEach(n => { if (teethData[n]) { teethData[n].missing = true; document.getElementById(`t-${n}`).classList.add('missing'); } }); speak(`紀錄完畢，${nums.join(", ")}號缺失`); } else if (nums) { // 如果聽到數字但冇關鍵字，比個提示 speak(`聽到數字 ${nums.join(" ")}, 請講埋缺失`); } }
