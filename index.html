// --- 語音辨識深度優化 --- const recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)(); recognition.lang = 'en-US'; recognition.continuous = true; recognition.interimResults = true; // 開啟即時修正 recognition.maxAlternatives = 1; recognition.onresult = (event) => { let interimTranscript = ''; let finalTranscript = ''; for (let i = event.resultIndex; i < event.results.length; ++i) { if (event.results[i].isFinal) { finalTranscript += event.results[i][0].transcript; } else { interimTranscript += event.results[i][0].transcript; } } // 在畫面上顯示「我現在聽到了什麼」 document.getElementById('log').innerHTML = `Listening: ${interimTranscript}
Confirmed: ${finalTranscript}`; const text = finalTranscript.toLowerCase(); // --- 靈敏的數字抓取邏輯 --- const nums = text.match(/\d+/g); // 1. 處理模式切換 if (text.includes("general")) { mode = "GENERAL"; speak("General mode"); return; } if (text.includes("periodontal") || text.includes("perio")) { mode = "PERIO"; speak("Perio mode"); return; } // 2. 處理 Missing (批次處理) if (mode === "GENERAL" && text.includes("missing") && nums) { nums.forEach(n => { const id = parseInt(n); if (teethData[id]) { teethData[id].missing = true; document.getElementById(`unit-${id}`).classList.add('missing'); } }); speak(`Marked ${nums.length} teeth missing`); } // 3. 處理 Perio 數值 (精準對位) if (mode === "PERIO" && nums && nums.length === 3) { const step = sequence[currentStep]; const id = step.teeth[currentIdx]; // 語音回饋與數據寫入 teethData[id][step.surface] = nums.join(""); document.getElementById(`v-${step.surface}-${id}`).innerText = nums.join(""); speak(`${id}, ${nums.join(" ")}`); currentIdx++; // 自動跳轉邏輯... updateHighlight(); } };
