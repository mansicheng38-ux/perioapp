document.getElementById('start-btn').onclick = async () => { try { alert("Debug: 按鈕已被觸發！"); // 第一步確認：按鈕有沒有壞掉 // 檢查瀏覽器是否支援語音 const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; if (!SpeechRecognition) { alert("錯誤：你的瀏覽器不支援語音辨識功能，請換成 Chrome 或 Safari。"); return; } // 檢查麥克風權限 alert("Debug: 正在請求麥克風權限..."); await navigator.mediaDevices.getUserMedia({ audio: true }); alert("Debug: 麥克風授權成功！"); // 啟動功能 recognition.start(); document.getElementById('start-btn').innerText = "Recording..."; document.getElementById('status').innerText = "Listening..."; speak("System online."); } catch (err) { // 如果失敗，會直接跳出錯誤代碼 alert("啟動失敗，錯誤訊息：\n" + err.name + ": " + err.message); console.error(err); } };
